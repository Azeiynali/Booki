                    var requestOptions = {
                        method: 'POST',
                        headers: myHeaders,
                        body: urlencoded,
                        redirect: 'follow'
                    };

                    fetch("http://127.0.0.1:1223/api/addMessage", requestOptions)
                    .then(response => response.text())

                    .then(result => {
                        this.$emit("new-message", this.message, JSON.parse(result).date, JSON.parse(result).id)
